SELECT 
    NON EMPTY 
        [DIM States].[State Name].MEMBERS *
        [DIM Counties].[County Name].MEMBERS ON ROWS,
    NON EMPTY 
        [DIM Date].[Year].MEMBERS *
        [DIM Date].[Day Light Saving].MEMBERS *
        [FACT Air Quality].[Category].MEMBERS ON COLUMNS
FROM 
    [DDS AIR 5]
WHERE 
    ([Measures].[AQI])
