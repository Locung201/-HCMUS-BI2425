WITH 
MEMBER [Measures].[DayCount] AS 
    COUNT(
        FILTER(
            [DIM Date].[Day].[Day].MEMBERS,
            [Measures].[AQI] > 0
        )
    )
SELECT 
    {[Measures].[DayCount]} ON COLUMNS,
    NON EMPTY 
    CROSSJOIN(
        [DIM States].[State Name].[State Name].MEMBERS,
		[DIM Counties].[County Name].[County Name].MEMBERS,
        [DIM Date].[Year].[Year].MEMBERS *
        [DIM Date].[Month].[Month].MEMBERS,
        [FACT Air Quality].[Category].[Category].MEMBERS
    ) ON ROWS
FROM [DDS AIR 1]
